template<auto Shaper>
template<auto OtherShaper>

constexpr kwk::shape< Shaper >::shape ( shape< OtherShaper > const &  other)
inlineconstexprnoexcept

Constructs a kwk::shape from a shape with higher ranks, filling the missing sizes with 1

This constructor does not participate in overload resolution of the shape is not fully specified at runtime.

Example

See it live on Compiler Explorer

//==================================================================================================
/*

#include <iostream>
#include <kwk/kwk.hpp>
int main()
{
// x will contains 4 x 32
kwk::shape< kwk::_2D > x(kwk::of_size(4, 8, 2, 2));
std::cout << x << "\n";
// y will contains 7 x 1 x 1
kwk::shape< kwk::_3D > y = kwk::of_size(7);
std::cout << y << "\n";
}
